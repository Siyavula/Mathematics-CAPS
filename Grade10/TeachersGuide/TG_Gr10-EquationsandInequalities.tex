\therefore V = \sqrt{\frac{V}{\pi h}}$
\item $E = \frac{hc}{\lambda}\\
E\lambda = hc\\
\therefore h= \frac{E\lambda}{c}$
\item $A=P(1+i)^n\\
\left(\frac{A}{P}\right)^{\frac{1}{h}} = (1+i)^{n \times \frac{1}{n}}\\
\left(\frac{A}{P}\right) - 1 = i\\
\therefore i = \left(\frac{A}{P}\right) - 1$

ft\lambda=D\\
